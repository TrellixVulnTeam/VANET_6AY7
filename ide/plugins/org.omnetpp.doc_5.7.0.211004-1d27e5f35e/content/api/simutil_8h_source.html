54 template<typename ToInt, typename FromInt>
55 ToInt checked_int_cast(FromInt x, const char *errmsg=nullptr)
56 {
57  static_assert(std::is_integral<ToInt>::value && std::is_integral<FromInt>::value, "checked_int_cast expects integers");
58  ToInt res = x;
59  if ((x<0) != (res<0) || x-res != 0) // note: x!=res would result in warning: signed-unsigned comparison
60  intCastError(std::to_string(x), errmsg);
61  return res;
62 }
